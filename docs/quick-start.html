Quick Start

Load & Read

use Xfa\Pdf\Facades\XfaPdf;

// Load a PDF
XfaPdf::setFile('/path/to/form.pdf');

// List all sections
$sections = XfaPdf::sections();
// ['CommonSections', 'Section_3_AppraisalDetails', ...]

// Read all fields in a section
$data = XfaPdf::read('Section_3_AppraisalDetails');
// ['appraisalYear' => '2024', 'appriseeName' => 'Dr Smith', ...]

// Read a single field
$year = XfaPdf::readField('Section_3_AppraisalDetails', 'appraisalYear');
// '2024'

Update & Save

// Update multiple fields at once
XfaPdf::update('Section_3_AppraisalDetails', [
    'appraisalYear' => '2025',
    'appriseeName'  => 'Dr Jane Smith',
    'contactEmail'  => 'jane@example.com',
]);

// Save changes (incremental update, preserves original PDF)
XfaPdf::save();

// Or save to a new file
XfaPdf::save('/path/to/output.pdf');

Fluent Section API

Use named section accessors for cleaner code:

// Read via section accessor
$details = XfaPdf::personalDetails()->read();
$year = XfaPdf::personalDetails()->readField('appraisalYear');

// Fluent updates
XfaPdf::personalDetails()
    ->update([
        'appraisalYear' => '2025',
        'appriseeName'  => 'Dr Smith',
    ]);

XfaPdf::cpd()
    ->updateField('commentaryOnActivities', 'Completed 50 hours of CPD');

XfaPdf::save();

Generate Preview

// Generate an HTML preview of the form
$html = XfaPdf::preview();

// Return as a response in a controller
return response($html);

Get Raw Data

// Get the raw XFA datasets XML
$xml = XfaPdf::rawXml();

// Convert entire form to a nested array
$allData = XfaPdf::toArray();

// Get field metadata (types, options, captions)
$meta = XfaPdf::fieldMetadata();

// Get repeatable subform definitions
$repeatables = XfaPdf::repeatableSubforms();

// Get navigation section labels
$navLabels = XfaPdf::navigationSections();
